⚙️ Generación automática del dataset
% Regla principal: crea el archivo CSV con datos simulados
crear_dataset :-
    open('dataset.csv', write, Stream),

    % Encabezado del archivo CSV
    write(Stream, 'Edad,Fuma,Alcohol,Ejercicio,Riesgo\n'),

    % Recorremos edades de 5 a 100
    forall(between(5,100,Edad), (

        % -------- RIESGO BAJO --------
        ( Edad < 30 -> EjB = 1, FumB = 0, AlcB = 0
        ; Edad < 60 -> EjB = 1, FumB = 0, AlcB = 1
        ; EjB = 0, FumB = 0, AlcB = 1
        ),
        format(Stream, '~w,~w,~w,~w,bajo~n', [Edad,FumB,AlcB,EjB]),

        % -------- RIESGO MEDIO --------
        ( Edad < 30 -> EjM = 1, FumM = 0, AlcM = 1
        ; Edad < 60 -> EjM = 0, FumM = 1, AlcM = 1
        ; EjM = 0, FumM = 1, AlcM = 1
        ),
        format(Stream, '~w,~w,~w,~w,medio~n', [Edad,FumM,AlcM,EjM]),

        % -------- RIESGO ALTO --------
        EjA = 0, FumA = 1, AlcA = 1,
        format(Stream, '~w,~w,~w,~w,alto~n', [Edad,FumA,AlcA,EjA])
    )),

    % Cerramos archivo
    close(Stream),

    % Mensaje en consola
    writeln('Archivo dataset.csv generado con éxito.').
