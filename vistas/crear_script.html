Función de predicción
async function predecir() {

    const edad = Number(document.getElementById("edad")?.value);
    const fuma = Number(document.getElementById("fuma")?.value);
    const alcohol = Number(document.getElementById("alcohol")?.value);
    const ejercicio = Number(document.getElementById("ejercicio")?.value);

    if ([edad, fuma, alcohol, ejercicio].some(v => isNaN(v))) {
        mostrarMensaje("Completa todos los campos correctamente");
        return;
    }

    try {
        const res = await fetch("/predict", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ edad, fuma, alcohol, ejercicio })
        });

        const data = await res.json();

        if (!data || !data.prob) {
            mostrarMensaje("❌ Error en respuesta del servidor");
            return;
        }


        const prob = data.prob;

        const mayor = obtenerMayor(prob);

        mostrarMensaje(
            `Riesgo: ${data.riesgo}\n` +
            `Puntaje: ${data.puntaje.toFixed(2)}\n` +
            `Dominante: ${mayor.tipo} (${mayor.valor}%)`
        );

        mostrarGrafica(prob);

    } catch (err) {
        console.error(err);
        mostrarMensaje("❌ Error en predicción");
    }
}

Determinación del valor dominante
function obtenerMayor(prob) {
    const arr = [
        { tipo: "Alto", valor: prob.alto },
        { tipo: "Medio", valor: prob.medio },
        { tipo: "Bajo", valor: prob.bajo }
    ];

    return arr.sort((a, b) => b.valor - a.valor)[0];
}

Convierte el objeto de probabilidades en un arreglo para poder ordenarlo.

Se utiliza sort() para ordenar de mayor a menor y se selecciona el primer elemento.

Esto permite identificar el riesgo dominante de forma sencilla.

Renderizado de gráfica
function mostrarGrafica(prob) {

    if (chart) chart.destroy();

    const ctx = document.getElementById("grafica").getContext("2d");

    chart = new Chart(ctx, {
        type: "bar",
        data: {
            labels: ["Alto", "Medio", "Bajo"],
            datasets: [{
                label: "Nivel de riesgo",
                data: [prob.alto, prob.medio, prob.bajo], // ✔ correcto
                backgroundColor: ["#ff0000", "#ffa500", "#008000"]
            }]
        },
        options: {
            scales: {
                y: {
                    beginAtZero: true,
                    max: 100
                }
            }
        }
    });
}

Genera una gráfica tipo barra con Chart.js utilizando los datos recibidos.

  • Se usa contexto 2D del canvas
  • Se configuran escalas y límites

La gráfica representa directamente la salida del backend.

function mostrarToast(mensaje, tipo = "success") {
    const toastEl = document.getElementById("toastMsg");

    // Mostrar el contenedor (porque estaba oculto)
    toastEl.style.display = "block";

    // Cambiar mensaje
    toastEl.querySelector(".toast-body").innerText = mensaje;

    // Cambiar color
    toastEl.className = `toast mt-2 text-bg-${tipo} border-0`;

    const toast = new bootstrap.Toast(toastEl, {
        delay: 3000
    });

    toast.show();
}

Subida de CSV
async function subirCSV() {
    const file = document.getElementById("fileInput")?.files?.[0];

    if (!file) {
        alert("Selecciona un archivo primero");
        return;
    }

    const formData = new FormData();
    formData.append("data", file);

    try {
        const res = await fetch("/upload", {
            method: "POST",
            body: formData
        });

        const data = await res.json();

        if (data?.status === "ok") {
            mostrarToast("CSV cargado correctamente ✅", "success");
        } else {
            mostrarToast("Error al subir archivo ❌", "danger");
        }

    } catch (err) {
        console.error(err);
        mostrarToast("Error en upload ❌", "danger");
    }
}

Envía el archivo CSV al backend utilizando FormData.
