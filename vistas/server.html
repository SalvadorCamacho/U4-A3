Subida de CSV
% =========================================================================
% MANEJO DE ARCHIVO CSV (UPLOAD)
% =========================================================================

upload_handler(Request) :-
    % catch/3 captura errores para evitar que el servidor se caiga si el CSV está mal formado.
    catch(
        (
            % Lee los datos crudos de la petición POST.
            http_read_data(Request, Data, []),
            % Busca dentro de los datos el campo 'data' (el contenido del archivo).
            ( member(data=CSVAtom, Data) ->
                atom_string(CSVAtom, CSVString),           % Convierte el átomo de Prolog a String.
                % Divide el string por saltos de línea. Omite la primera fila (cabecera).
                split_string(CSVString, "\n", "\r", [_|Filas]),
                retractall(dato(_,_,_,_,_)),               % Limpia la "base de datos" dinámica actual.
                maplist(procesar_fila, Filas),             % Ejecuta 'procesar_fila' para cada línea.
                reply_json(_{status:"ok"})                 % Responde al cliente con éxito.
            ; reply_json(_{status:"no_file"})              % Error si no se envió el campo 'data'.
            )
        ),
        _, % Si ocurre un error, se ignora la variable de error.
        reply_json(_{status:"error"})                      % Responde error genérico.
    ).

Procesamiento de filas
% Predicado para procesar líneas vacías (evita errores al final del archivo).
procesar_fila("") :- !.
% Procesa una línea de texto separada por comas.
procesar_fila(Fila) :-
    split_string(Fila, ",", "", [E,F,A,Ej,RStr]), % Divide por comas en variables temporales.
    number_string(Edad, E),                        % Convierte texto a número.
    number_string(Fuma, F),
    number_string(Alcohol, A),
    number_string(Ejercicio, Ej),
    atom_string(Riesgo, RStr),                     % Convierte texto a átomo de Prolog.
    % Agrega el nuevo hecho a la base de datos dinámica.
    assert(dato(Edad, Fuma, Alcohol, Ejercicio, Riesgo)).

Convierte cada fila en un hecho lógico.

Probabilidades
% =========================================================================
% LÓGICA DE NEGOCIO (PROBABILIDADES)
% =========================================================================

probabilidades_seguras(P, Alto, Medio, Bajo) :-
    % Is es para evaluación aritmética. max/min aseguran que esté entre 0 y 100.
    P1 is max(0, min(100, P)),

    % Estructura condicional: ( Condición -> Si_Verdadero ; Si_Falso ).
    ( P1 >= 60 ->
        Alto = 80, Medio = 15, Bajo = 5
    ; P1 >= 40 ->
        Alto = 20, Medio = 60, Bajo = 20
    ;
        Alto = 10, Medio = 20, Bajo = 70
    ).

Genera probabilidades basadas en reglas.

Predicción
% =========================================================================
% PREDICCIÓN (API JSON)
% =========================================================================

predict_handler(Request) :-
    catch(
        (
            % Obtiene el cuerpo de la petición POST directamente como un Diccionario de Prolog.
            http_read_json_dict(Request, D),

            % Accede a los valores del diccionario JSON enviado por el cliente.
            Edad = D.get(edad),
            Fuma = D.get(fuma),
            Alcohol = D.get(alcohol),
            Ejercicio = D.get(ejercicio),

            % Forzar conversión a flotante para cálculos aritméticos precisos.
            E is float(Edad),
            F is float(Fuma),
            A is float(Alcohol),
            J is float(Ejercicio),

            % Fórmula matemática para calcular el puntaje de riesgo.
            Puntaje is E*0.2 + F*35 + A*15 - J*25,

            % Lógica de clasificación basada en el puntaje.
            ( Puntaje > 60 -> R = alto
            ; Puntaje > 40 -> R = medio
            ; R = bajo
            ),

            % Llama a la función de cálculo de probabilidades definida arriba.
            probabilidades_seguras(Puntaje, Alto, Medio, Bajo),

            % Responde enviando un objeto JSON construido al vuelo.
            reply_json(_{
                riesgo:R,
                puntaje:Puntaje,
                prob:_{alto:Alto, medio:Medio, bajo:Bajo}
            })
        ),
        Error,
        (
            % En caso de excepción, convierte el error a texto y lo envía en el JSON.
            message_to_string(Error, Msg),
            reply_json(_{error:Msg})
        )
    ).
