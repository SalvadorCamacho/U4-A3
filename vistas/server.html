upload_handler(Request) :-
    catch(
        (
            http_read_data(Request, Data, []),
            ( member(data=CSVAtom, Data) ->
                atom_string(CSVAtom, CSVString),
                split_string(CSVString, "\n", "\r", [_|Filas]),
                retractall(dato(_,_,_,_,_)),
                maplist(procesar_fila, Filas),
                reply_json(_{status:"ok"})
            ;   reply_json(_{status:"no_file"})
            )
        ),
        _,
        reply_json(_{status:"error"})
    ).

procesar_fila("") :- !.
procesar_fila(Fila) :-
    split_string(Fila, ",", "", [E,F,A,Ej,RStr]),
    number_string(Edad, E),
    number_string(Fuma, F),
    number_string(Alcohol, A),
    number_string(Ejercicio, Ej),
    atom_string(Riesgo, RStr),
    assert(dato(Edad, Fuma, Alcohol, Ejercicio, Riesgo)).

¿Qué hace este bloque?

Convierte un archivo CSV en hechos Prolog almacenados dinámicamente.

  • Lectura de archivos
  • Transformación a datos lógicos
  • Uso de assert/retract

predict_handler(Request) :-
    http_read_json_dict(Request, D),
    Puntaje is D.edad*0.2 + D.fuma*35 + D.alcohol*15 - D.ejercicio*25,
    
    ( Puntaje > 65 -> R = alto
    ; Puntaje > 35 -> R = medio
    ; R = bajo

stats_handler(_) :-
    findall(E, dato(E,_,_,_,_), Edades),
    ( Edades == [] ->
        reply_json(_{error:"No hay datos"})
    ;   promedio(Edades, Prom),
        conteo(alto, A),
        conteo(medio, M),
        conteo(bajo, B),
        reply_json(_{
            promedioEdad: Prom,
            riesgo: _{alto:A, medio:M, bajo:B}
        })
    ).

promedio(L, P) :- sum_list(L, S), length(L, N), P is S / N.
conteo(R, N) :- findall(1, dato(_,_,_,_,R), L), length(L, N).

¿Qué hace este bloque?

Genera estadísticas a partir de los datos almacenados.

  • Promedio de edades
  • Conteo por categoría
  • Análisis en tiempo real
